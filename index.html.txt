            };
            const loadState = () => {
                const saved = localStorage.getItem(CONSTANTS.STORAGE_KEY);
                if (saved) {
                    try {
                        const parsed = JSON.parse(saved);
                        state = { ...state, ...parsed };
                        if (state.expectedShots.length < state.targetCount) {
                            state.expectedShots = Array.from({length: state.targetCount}, (_, i) => state.expectedShots[i] || 5);
                        }
                    } catch (e) { console.error(e); }
                } else {
                    state.expectedShots = Array(state.targetCount).fill(5);
                }
            };
            const saveState = () => {
                localStorage.setItem(CONSTANTS.STORAGE_KEY, JSON.stringify(state));
                calculateStats();
                if (state.globalReceived > 0) els.settingsDot.classList.add('hidden');
            };
            // --- Logic & Calc ---
            const calculateStats = () => {
                let totalSent = 0, totalReceived = 0, totalUnfired = 0, totalMisfires = 0, totalDeformed = 0;
                state.waves.forEach(wave => {
                    wave.targets.forEach(t => {
                        totalSent += (parseInt(t.sent) || 0);
                        totalReceived += (parseInt(t.recv) || 0);
                        totalUnfired += (parseInt(t.unfired) || 0);
                        totalMisfires += (parseInt(t.misfires) || 0);
                        totalDeformed += (parseInt(t.deformed) || 0);
                    });
                });
                const actualConsumed = totalSent - totalUnfired;
                const remaining = state.globalReceived - actualConsumed;
                // Update Header
                els.remainingDiv.textContent = remaining;
                els.remainingDiv.className = `text-2xl font-bold leading-none tabular-nums ${remaining < 0 ? 'text-red-500' : 'text-emerald-600'}`;
                // Waves Calc
                const shotsPerWave = state.expectedShots.reduce((a, b) => a + (parseInt(b)||0), 0);
                if (shotsPerWave > 0 && remaining > 0) {
                    const fullWaves = Math.floor(remaining / shotsPerWave);
                    const remainder = remaining % shotsPerWave;
                    els.remainingWavesDiv.textContent = fullWaves;
                    if (remainder > 0) {
                        els.partialWaveSpan.textContent = `+${remainder}`;
                        els.partialWaveSpan.classList.remove('hidden');
                    } else {
                        els.partialWaveSpan.classList.add('hidden');
                    }
                } else {
                    els.remainingWavesDiv.textContent = remaining > 0 ? "∞" : "0";
                    els.partialWaveSpan.classList.add('hidden');
                }
                // Update Footer & Modal
                els.totalConsumed.textContent = actualConsumed;
                els.totalUnfired.textContent = totalUnfired;
                els.modalUnfired.textContent = totalUnfired;
                els.modalMisfires.textContent = totalMisfires;
                els.modalDeformed.textContent = totalDeformed;
                els.modalTotalSent.textContent = totalSent;
                els.modalTotalRecv.textContent = totalReceived;
                els.modalConsumed.textContent = actualConsumed;
                updateRowCalculations();
            };
            // --- Rendering ---
            const renderUI = () => {
                els.globalInput.value = state.globalReceived || '';
                els.targetInput.value = state.targetCount;
                renderExpectedInputs();
                renderTable();
                calculateStats();
            };
            const renderExpectedInputs = () => {
                els.expectedContainer.innerHTML = state.expectedShots.map((val, i) => `
                    <div class="flex-shrink-0 w-12 text-center">
                        <label class="text-[10px] text-gray-400 block mb-1">#${i+1}</label>
                        <input type="number" value="${val}" 
                            onchange="app.updateExpected(${i}, this.value)"
                            class="w-full text-center settings-input border border-gray-300 rounded text-sm py-1 bg-white" placeholder="-">
                    </div>
                `).join('');
            };
            const renderTable = () => {
                // Header
                let headerRow = `<th class="sticky-col z-30 w-10 text-center shadow-sm text-gray-400">#</th>`;
                for (let i = 0; i < state.targetCount; i++) {
                    headerRow += `
                        <th class="min-w-[50px] border-l border-gray-100">
                            <div class="col-header-icon text-blue-600"><i class="fas fa-paper-plane"></i><span>送</span></div>
                        </th>
                        <th class="min-w-[50px]">
                            <div class="col-header-icon text-green-600"><i class="fas fa-hand-holding"></i><span>收</span></div>
                        </th>
                        <th class="min-w-[40px] bg-slate-50">
                            <div class="col-header-icon text-gray-400"><i class="fas fa-ban"></i><span>未</span></div>
                        </th>
                        <th class="min-w-[40px] bg-slate-50">
                            <div class="col-header-icon text-red-400"><i class="fas fa-bolt"></i><span>壞</span></div>
                        </th>
                        <th class="min-w-[40px] bg-slate-50 border-r border-gray-200">
                            <div class="col-header-icon text-orange-400"><i class="fas fa-shapes"></i><span>變</span></div>
                        </th>
                    `;
                }
                els.tableHead.innerHTML = `<tr>${headerRow}</tr>`;
                // Body
                if (state.waves.length === 0) {
                    els.tableBody.innerHTML = `<tr><td colspan="${1 + state.targetCount * 5}" class="py-12 text-center text-gray-400 text-sm">點擊右下角「新增波次」開始</td></tr>`;
                    return;
                }
                els.tableBody.innerHTML = state.waves.map((wave, wIdx) => {
                    let rowHtml = `
                        <th class="sticky-col bg-white text-gray-500 font-medium text-xs shadow-[2px_0_5px_rgba(0,0,0,0.02)]">
                            <div class="flex flex-col items-center justify-center h-full gap-2">
                                <span class="text-sm font-bold text-gray-700">${wave.id}</span>
                                <button onclick="app.deleteWave(${wIdx})" class="text-gray-300 hover:text-red-500 p-2"><i class="fas fa-times"></i></button>
                            </div>
                        </th>
                    `;
                    wave.targets.forEach((t, tIdx) => {
                        // 差異計算 (生成時不計算，由 updateRowCalculations 統一處理)
                        rowHtml += `
                            <td class="border-l border-gray-100 relative group">
                                <input type="number" value="${t.sent}" oninput="app.updateCell(${wIdx}, ${tIdx}, 'sent', this.value)" placeholder="-">
                            </td>
                            <td class="relative">
                                <input type="number" value="${t.recv}" oninput="app.updateCell(${wIdx}, ${tIdx}, 'recv', this.value)" placeholder="-" class="text-green-700 font-medium">
                                <span id="diff-tag-${wIdx}-${tIdx}" class="absolute top-0 right-0 pointer-events-none"></span>
                            </td>
                            <td class="bg-slate-50"><input type="number" value="${t.unfired}" oninput="app.updateCell(${wIdx}, ${tIdx}, 'unfired', this.value)" placeholder="" tabindex="-1" class="text-gray-500 text-xs"></td>
                            <td class="bg-slate-50"><input type="number" value="${t.misfires}" oninput="app.updateCell(${wIdx}, ${tIdx}, 'misfires', this.value)" placeholder="" tabindex="-1" class="text-red-400 text-xs"></td>
                            <td class="bg-slate-50 border-r border-gray-200"><input type="number" value="${t.deformed}" oninput="app.updateCell(${wIdx}, ${tIdx}, 'deformed', this.value)" placeholder="" tabindex="-1" class="text-orange-400 text-xs"></td>
                        `;
                    });
                    return `<tr>${rowHtml}</tr>`;
                }).join('');
            };
            const updateRowCalculations = () => {
                state.waves.forEach((wave, wIdx) => {
                    wave.targets.forEach((t, tIdx) => {
                        const tag = document.getElementById(`diff-tag-${wIdx}-${tIdx}`);
                        if (tag) {
                            const sent = parseInt(t.sent) || 0;
                            const recv = parseInt(t.recv) || 0;
                            const unfired = parseInt(t.unfired) || 0;
                            const diff = sent - recv - unfired;
                            if (diff === 0) {
                                tag.innerHTML = '';
                            } else {
                                const isLack = diff > 0; // 缺
                                tag.innerHTML = `<div class="text-[10px] px-1 py-0.5 ${isLack ? 'text-white bg-red-500' : 'text-white bg-blue-500'} rounded-bl opacity-90 shadow-sm">${isLack ? '-'+diff : '+'+Math.abs(diff)}</div>`;
                            }
                        }
                    });
                });
            };
            // --- Interactions ---
            const toggleSettings = (forceOpen = false) => {
                const isOpen = els.settingsPanel.classList.contains('open');
                if (forceOpen || !isOpen) {
                    els.settingsPanel.classList.add('open');
                    document.getElementById('settingsBtn').classList.add('text-blue-600', 'bg-blue-50');
                } else {
                    els.settingsPanel.classList.remove('open');
                    document.getElementById('settingsBtn').classList.remove('text-blue-600', 'bg-blue-50');
                }
            };
            // Generic Toggle for Modals
            const toggleModal = (modalId, overlayId, contentId) => {
                const modal = document.getElementById(modalId);
                const overlay = document.getElementById(overlayId);
                const content = document.getElementById(contentId);
                const isHidden = modal.classList.contains('hidden');
                if (isHidden) {
                    modal.classList.remove('hidden');
                    requestAnimationFrame(() => {
                        overlay.classList.remove('opacity-0');
                        content.classList.remove('translate-y-full');
                    });
                } else {
                    overlay.classList.add('opacity-0');
                    content.classList.add('translate-y-full');
                    setTimeout(() => modal.classList.add('hidden'), 300);
                }
            };
            const toggleStats = () => toggleModal('statsModal', 'statsOverlay', 'statsContent');
            const toggleHelp = () => toggleModal('helpModal', 'helpOverlay', 'helpContent');
            const updateGlobal = (val) => {
                state.globalReceived = parseInt(val) || 0;
                saveState();
            };
            const adjustTargets = (delta) => {
                let newVal = state.targetCount + delta;
                if (newVal < 1) newVal = 1;
                if (newVal > CONSTANTS.MAX_TARGETS) newVal = CONSTANTS.MAX_TARGETS;
                state.targetCount = newVal;
                if (state.expectedShots.length < newVal) {
                    for(let i=state.expectedShots.length; i<newVal; i++) state.expectedShots.push(5);
                }
                state.waves.forEach(wave => {
                    if (wave.targets.length < newVal) {
                        for(let i=wave.targets.length; i<newVal; i++) {
                            wave.targets.push({ sent: state.expectedShots[i], recv: '', unfired: '', misfires: '', deformed: '' });
                        }
                    } else {
                        wave.targets.length = newVal;
                    }
                });
                els.targetInput.value = newVal;
                saveState();
                renderUI();
            };
            const updateExpected = (idx, val) => {
                state.expectedShots[idx] = parseInt(val) || 0;
                saveState();
            };
            const addWave = () => {
                const newId = state.waves.length > 0 ? state.waves[state.waves.length - 1].id + 1 : 1;
                const newTargets = Array.from({ length: state.targetCount }, (_, i) => ({
                    sent: state.expectedShots[i] || 0,
                    recv: '',
                    unfired: '',
                    misfires: '',
                    deformed: ''
                }));
                state.waves.push({ id: newId, targets: newTargets });
                saveState();
                renderTable();
                setTimeout(() => {
                    const wrapper = document.getElementById('tableWrapper');
                    wrapper.scrollTo({ top: wrapper.scrollHeight, behavior: 'smooth' });
                }, 100);
            };
            const updateCell = (wIdx, tIdx, field, val) => {
                state.waves[wIdx].targets[tIdx][field] = val;
                calculateStats();
                if (app.saveTimer) clearTimeout(app.saveTimer);
                app.saveTimer = setTimeout(() => {
                    localStorage.setItem(CONSTANTS.STORAGE_KEY, JSON.stringify(state));
                }, 500);
            };
            // Modal Confirm System
            let confirmCallback = null;
            const openConfirm = (title, msg, cb) => {
                els.modalTitle.textContent = title;
                els.modalMsg.textContent = msg;
                confirmCallback = cb;
                els.modalOverlay.classList.remove('hidden');
                requestAnimationFrame(() => {
                    els.modalOverlay.classList.remove('opacity-0');
                    els.modalBox.classList.remove('scale-95');
                });
            };
            const closeModal = () => {
                els.modalOverlay.classList.add('opacity-0');
                els.modalBox.classList.add('scale-95');
                setTimeout(() => els.modalOverlay.classList.add('hidden'), 200);
                confirmCallback = null;
            };
            els.modalConfirmBtn.onclick = () => {
                if (confirmCallback) confirmCallback();
                closeModal();
            };
            const deleteWave = (idx) => {
                openConfirm('刪除波次', `確定要刪除第 ${state.waves[idx].id} 波次嗎？`, () => {
                    state.waves.splice(idx, 1);
                    // Reset IDs? Keep logical for now
                    // state.waves.forEach((w, i) => w.id = i + 1);
                    saveState();
                    renderTable();
                });
            };
            const resetData = () => {
                openConfirm('重置資料', '確定要清空所有紀錄嗎？設定將保留。', () => {
                    state.waves = [];
                    saveState();
                    renderTable();
                });
            };
            const exportData = () => {
                const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(state));
                const a = document.createElement('a');
                a.href = dataStr;
                a.download = `range_data_${new Date().toISOString().slice(0,10)}.json`;
                a.click();
            };
            return {
                init,
                toggleSettings,
                toggleStats,
                toggleHelp,
                updateGlobal,
                adjustTargets,
                updateExpected,
                addWave,
                deleteWave,
                updateCell,
                resetData,
                closeModal,
                exportData
            };
        })();
        window.onload = app.init;
    </script>
</body>
</html>
